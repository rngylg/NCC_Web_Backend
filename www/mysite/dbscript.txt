use myshop;

create table original_place(
    pid int primary key auto_increment,
    source VARCHAR(100),
    vendor VARCHAR(255)
);

create table products (
    pid int primary key auto_increment,
    name VARCHAR(100) not null,
    price float not null, 
    photo VARCHAR(255) not null,
    sid int,
    foreign key(sid) references original_place(pid)
);
